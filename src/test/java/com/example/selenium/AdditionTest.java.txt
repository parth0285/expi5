package com.example.tests;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.edge.EdgeDriver;
import org.openqa.selenium.edge.EdgeOptions;
import org.testng.Assert;
import org.testng.annotations.AfterMethod;
import org.testng.annotations.BeforeMethod;
import org.testng.annotations.Test;

public class AdditionTest {

    WebDriver driver;

    @BeforeMethod
    public void setup() {

        EdgeOptions options = new EdgeOptions();
        options.addArguments("--start-maximized");

        // Selenium Manager automatically handles driver
        driver = new EdgeDriver(options);

        driver.get("file:///C:/Users/YourName/Desktop/addition.html");
    }

    @Test
    public void testAddition() {

        driver.findElement(By.id("num1")).sendKeys("15");
        driver.findElement(By.id("num2")).sendKeys("25");
        driver.findElement(By.tagName("button")).click();

        String result = driver.findElement(By.id("result")).getText();

        Assert.assertEquals(result, "Sum is: 40");
    }

    @AfterMethod
    public void tearDown() {
        driver.quit();
    }
}
